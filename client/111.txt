// --- 优雅的滚动浮现 Hook ---
const useScrollReveal = (threshold = 0.1) => {
  const ref = useRef(null);
  const [isVisible, setIsVisible] = useState(false);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          observer.unobserve(entry.target);
        }
      },
      { threshold, rootMargin: "0px 0px -50px 0px" }
    );
    if (ref.current) observer.observe(ref.current);
    return () => observer.disconnect();
  }, [threshold]);

  return [ref, isVisible];
};

// --- 渐显包裹组件 ---
const FadeIn = ({ children, delay = 0, className = "" }) => {
  const [ref, isVisible] = useScrollReveal();
  return (
    <div 
      ref={ref} 
      className={`transition-all duration-1000 ease-[cubic-bezier(0.16,1,0.3,1)] ${
        isVisible ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-8'
      } ${className}`}
      style={{ transitionDelay: `${delay}ms` }}
    >
      {children}
    </div>
  );
};

// --- 交互式有机呼吸背景组件 ---
const InteractiveBackground = () => {
  useEffect(() => {
    const blob1 = document.getElementById('blob1');
    const blob2 = document.getElementById('blob2');
    const blob3 = document.getElementById('blob3');

    // 默认引力中心在屏幕正中央
    let targetX = window.innerWidth / 2;
    let targetY = window.innerHeight / 2;
    let currentX = targetX;
    let currentY = targetY;

    // 监听鼠标点击事件，改变引力中心
    const handleGlobalClick = (e) => {
      targetX = e.clientX;
      targetY = e.clientY;
    };
    window.addEventListener('click', handleGlobalClick);

    let animationFrameId;
    const render = () => {
      const time = Date.now();

      // 缓动算法：让当前中心点丝滑地向鼠标点击点移动 (引力效果)
      currentX += (targetX - currentX) * 0.03;
      currentY += (targetY - currentY) * 0.03;

      const updateBlob = (el, size, radiusX, radiusY, speed, offsetX, offsetY) => {
        if (!el) return;
        // 使用三角函数生成自由、不规则的游荡轨迹
        const x = currentX + Math.sin(time * speed + offsetX) * radiusX - size / 2;
        const y = currentY + Math.cos(time * speed * 0.8 + offsetY) * radiusY - size / 2;
        el.style.transform = `translate(${x}px, ${y}px)`;
      };

      // 参数：元素, 尺寸, X游荡半径, Y游荡半径, 游荡速度, X相位, Y相位
      updateBlob(blob1, 600, 250, 300, 0.0005, 0, 0);
      updateBlob(blob2, 500, 350, 200, 0.0004, Math.PI, Math.PI / 2);
      updateBlob(blob3, 550, 200, 350, 0.0006, Math.PI / 2, Math.PI);

      animationFrameId = requestAnimationFrame(render);
    };
    render();

    return () => {
      window.removeEventListener('click', handleGlobalClick);
      cancelAnimationFrame(animationFrameId);
    };
  }, []);

  return (
    <div className="fixed inset-0 z-0 pointer-events-none overflow-hidden opacity-50 mix-blend-multiply">
      <div id="blob1" className="absolute top-0 left-0 w-[600px] h-[600px] bg-[#E0E7FF] blur-[100px] blob-shape-1 will-change-transform"></div>
      <div id="blob2" className="absolute top-0 left-0 w-[500px] h-[500px] bg-[#FCE7F3] blur-[120px] blob-shape-2 will-change-transform"></div>
      <div id="blob3" className="absolute top-0 left-0 w-[550px] h-[550px] bg-[#E0F2FE] blur-[100px] blob-shape-3 will-change-transform"></div>
    </div>
  );
};
